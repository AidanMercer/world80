# packages.txt — packages this rice depends on.
# Install repo packages with pacman, AUR packages with paru.
#   sudo pacman -S --needed $(grep -v '^#' packages.txt | sed -n '/^\[repo\]/,/^\[aur\]/p' | grep -v '^\[')
#   paru -S --needed       $(grep -v '^#' packages.txt | sed -n '/^\[aur\]/,$p'        | grep -v '^\[')
# (or just read it by eye — it's a plain list.)
#
# hardware/boot packages (nvidia, grub, ucode, mesa, linux) are left out on
# purpose — those come with the install, not the rice.
